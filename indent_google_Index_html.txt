  <script>
    // Add new product row
    function addRow() {
      const table = document.getElementById("indentContainer");
      const newRow = table.insertRow();
      newRow.innerHTML = `
        <td><input type="text" name="productCode" required></td>
        <td><input type="text" name="materialName" required></td>
        <td><input type="text" name="supplier" required></td>
        <td><input type="number" name="quantity" required></td>
        <td><input type="text" name="uom" required></td>
        <td><input type="text" name="remarks"></td>
        <td>
          <a href="https://docs.google.com/forms/d/e/1FAIpQLScKuuw4Q_KiEH_WDe8a9y-kc9WPccT_3H-4tPawGWLrZBKGHw/viewform" target="_blank">
            Upload Photo
          </a>
        </td>
        <td><button type="button" class="remove" onclick="removeRow(this)">X</button></td>
      `;
    }

    // Remove a row
    function removeRow(button) {
      button.closest("tr").remove();
    }

    // Submit form data to Apps Script
    function submitForm() {
      const form = document.getElementById("poForm");
      const formData = new FormData(form);

      const indents = [];
      document.querySelectorAll("#indentContainer tr").forEach(row => {
        const indent = {
          productCode: row.querySelector("[name='productCode']").value,
          materialName: row.querySelector("[name='materialName']").value,
          supplier: row.querySelector("[name='supplier']").value,
          quantity: row.querySelector("[name='quantity']").value,
          uom: row.querySelector("[name='uom']").value,
          remarks: row.querySelector("[name='remarks']").value
        };
        indents.push(indent);
      });

      const data = {
        styleId: formData.get('styleId'),
        client: formData.get('client'),
        poNumber: formData.get('poNumber'),
        poDate: formData.get('poDate'),
        indents: indents
      };

      google.script.run.withSuccessHandler(response => {
        document.getElementById("response").innerText = response;
        form.reset();
        document.getElementById("indentContainer").innerHTML = '';
        addRow();
      }).withFailureHandler(error => {
        document.getElementById("response").innerText = "Error: " + error.message;
      }).submitIndent(data);
    }
  </script>
